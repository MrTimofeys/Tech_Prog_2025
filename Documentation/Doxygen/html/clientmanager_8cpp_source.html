336void ClientManager::processResponse(const QJsonObject& response)
337{
338 QString message = response["message"].toString();
339
340 if (message.contains("Verification code sent")) {
341 storedVerificationCode = response["code"].toString();
342 emit verificationCodeResult(true, message);
343 } else if (message.contains("Password updated successfully")) {
344 storedVerificationCode.clear();
345 emit passwordResetResult(true, message);
346 } else if (message.contains("registered successfully")) {
347 emit registrationResult(true, message);
348 } else if (message.contains("Error")) {
349 if (message.contains("verification code")) {
350 emit verificationCodeResult(false, message);
351 } else if (message.contains("password")) {
352 emit passwordResetResult(false, message);
353 } else {
354 emit registrationResult(false, message);
355 }
356 } else if (message.contains("Login successful")) {
357 emit loginResult(true, message);
358 } else if (message.contains("Invalid")) {
359 emit loginResult(false, message);
360 } else if (response.contains("root")) {
361 double root = response["root"].toDouble();
362 message = response["message"].toString();
363 qDebug() << "Root found:" << root << "-" << message;
364 emit equationSolved(true, root, message); // Предполагается, что вы создадите такой сигнал
365 } else if (response["message"].toString().contains("No convergence")) {
366 emit equationSolved(false, 0.0, response["message"].toString());
